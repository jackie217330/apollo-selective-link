```bash
npm install --save apollo-selective-link
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'apollo-selective-link'
import 'apollo-selective-link/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
